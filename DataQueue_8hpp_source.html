50 template<typename T>
51 class DataQueue {
52 private:
53  T* values;
54  uint8_t head;
55  uint8_t size;
56 public:
57  explicit DataQueue(uint8_t size): size(size), head(0) {
58  if (size == 0) {
59  throw std::invalid_argument("Size must be at least 1");
60  }
61 
62  values = new T[size];
63  }
64 
65  virtual ~DataQueue() {
66  delete[] values;
67  }
68 
72  [[nodiscard]] uint8_t getSize() const { return size; }
73 
77  [[nodiscard]] uint8_t getNumberOfElements() const { return head; }
78 
84  [[nodiscard]] T* getValues() const { return values; }
85 
89  [[nodiscard]] T& getLatestValue() const noexcept(false) {
90  if (head == 0) {
91  throw std::out_of_range("DataQueue is empty.");
92  }
93 
94  return values[head - 1];
95  }
96 
102  void add(T value) {
103  if (head == size) {
104  for (uint8_t i = 1; i < size; i++) {
105  values[i - 1] = values[i];
106  }
107  head--;
108  }
109 
110  values[head] = value;
111  head++;
112  }
113 
120  void update(T value) {
121  if (head == 0) {
122  add(value);
123  return;
124  }
125 
126  uint8_t i = head - 1;
127  values[i] = value;
128  }
129 
130  using iterator = DataQueueIterator<T>;
131 
132  iterator begin() const {
133  return iterator(*this, 0);
134  }
135 
136  iterator end() const {
137  return iterator(*this, getNumberOfElements());
