14 template<typename... Ts>
15 class CombineLatest {
16 public:
17  using Callback = std::function<void(const Ts& ...)>;
18 
19  // Templated constructor to deduce the type of the lambda directly
20  template<typename Func>
21  explicit CombineLatest(Func&& callback, ObservedObject<Ts>& ... observedObjects)
22  : callback(std::forward<Func>(callback)), objects(observedObjects...) {
23  (observedObjects.addListener(std::bind(&CombineLatest::handleUpdate, this)), ...);
24  }
25 
26  void handleUpdate() {
27  auto latestValues = getLatestValues(objects, std::index_sequence_for<Ts...>{});
28  if (std::apply([](const auto& ... args) { return (args && ...); }, latestValues)) {
29  callCallback(latestValues);
30  }
31  }
32 
33 private:
34  // Helper function to get the latest values from ObservedObjects
35  template<std::size_t... Is>
36  static std::tuple<std::optional<Ts>...>
37  getLatestValues(const std::tuple<ObservedObject<Ts>& ...>& objects, std::index_sequence<Is...>) {
38  return std::make_tuple(std::get<Is>(objects).get()...);
39  }
40 
41  // Helper function to call the callback function with the latest values
42  template<std::size_t... Is>
43  void callCallback(const std::tuple<std::optional<Ts>...>& values, std::index_sequence<Is...>) {
44  callback(std::get<Is>(values).value()...);
45  }
46 
47  // Call the callback function with the latest values
48  void callCallback(const std::tuple<std::optional<Ts>...>& values) {
49  callCallback(values, std::index_sequence_for<Ts...>{});
50  }
51 
52  Callback callback;
53  std::tuple<ObservedObject<Ts>& ...> objects;
54 };
